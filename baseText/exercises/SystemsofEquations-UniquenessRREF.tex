\begin{ex}
  The following are augmented matrices for four systems of
  equations. Determine which of them, if any, are row equivalent.
  \begin{equation*}
    \begin{mymatrix}{rrrr|r}
       1 & 3 &  5 &  1 & 12 \\
      -1 & 1 & -1 &  2 &  5 \\
       2 & 0 &  4 & -2 &  0
    \end{mymatrix}
    \quad
    \begin{mymatrix}{rrrr|r}
      1 & 2 & 4 & -1 &  8 \\
      2 & 4 & 3 &  1 & 15 \\
      3 & 6 & 1 & -1 &  8
    \end{mymatrix}
    \quad
    \begin{mymatrix}{rrrr|r}
      3 & 6 & -3 & 1 & 6 \\
      2 & 4 & 2 & 1 & 13 \\
      0 & 0 & 1 & 0 & 2
    \end{mymatrix}
    \quad
    \begin{mymatrix}{rrrr|r}
      1 & 2 & 4 & 1 & 10 \\
      0 & 1 & 1 & 1 & 5 \\
      2 & 1 & 5 & 0 & 8
    \end{mymatrix}
  \end{equation*}

\begin{ex}
  Find a tuple $(x,y,z)$ that is a solution to one system of equations
  but not the other.
  \begin{equation*}
    \begin{mymatrix}{rrrr|r}
      1 & 0 & 1 & 5 & 0 \\
      2 & 1 & 4 & 3 & 0 \\
      3 & 1 & 5 & 9 & 0 \\
    \end{mymatrix}
    \quad
    \begin{mymatrix}{rrrr|r}
      1 & 0 & 3 & 6 & 0 \\
      2 & 1 & 8 & 2 & 0 \\
      2 & 0 & 6 & 13 & 0 \\
    \end{mymatrix}
  \end{equation*}    
\end{ex}
