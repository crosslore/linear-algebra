\section*{Exercises}


\begin{ex}
  Let
  \begin{equation*}
    A = \begin{mymatrix}{rr}
      1 & 3 \\
      5 & 2 \\
    \end{mymatrix},
    \quad
    B = \begin{mymatrix}{rrr}
      1 & 0 & 2 \\
      0 & 2 & 1 \\
    \end{mymatrix},
    \quad
    C = \begin{mymatrix}{rrr}
      2 & 1 & 0 \\
      1 & 0 & 2 \\
      0 & 3 & 1 \\
    \end{mymatrix}
  \end{equation*}
  and compute the following over $\Z_{11}$:
  \begin{enumerate}
  \item $3A$,
  \item $A^2$,
  \item $AB$,
  \item $BC$,
  \item $C^{-1}$.
  \end{enumerate}
\end{ex}
