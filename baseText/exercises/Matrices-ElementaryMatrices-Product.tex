\begin{ex}
  Write each of the following matrices as a product of elementary
  matrices, if possible, or else say why it is not possible.
  \begin{equation*}
    (a)~
    \begin{mymatrix}{rr}
      1 & 3 \\
      0 & 2 \\
    \end{mymatrix},
    \quad
    (b)~
    \begin{mymatrix}{rr}
      2 & 1 \\
      5 & 3 \\
    \end{mymatrix},
    \quad
    (c)~
    \begin{mymatrix}{rrr}
      1 & 2 & 0 \\
      0 & 1 & 3 \\
      2 & 4 & 1 \\
    \end{mymatrix},
    \quad
    (d)~
    \begin{mymatrix}{rrr}
      0 & 1 & 2 \\
      1 & 3 & 7 \\
      1 & 1 & 3 \\
    \end{mymatrix}.
  \end{equation*}
\end{ex}
