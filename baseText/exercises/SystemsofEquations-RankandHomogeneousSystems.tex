\section*{Exercises}


\begin{ex}
  Find the basic solutions of each of the following homogeneous
  systems of equations.
  \begin{equation*}
    (a)~
    \begin{array}{r@{~}c@{~}l}
      2x + 3y + 4z &=& 0 \\
      x - 2y + z &=& 0 \\
      4x - y + 6z &=& 0
    \end{array}
    \quad
    (b)~
    \begin{array}{r@{~}c@{~}l}
       x -  y +  z &=& 0 \\
      -x - 2y - 4z &=& 0 \\
      2x +  y + 5z &=& 0
    \end{array}
    \quad
    (c)~
    \begin{array}{r@{~}c@{~}l}
      x +  y -  z + 2w &=& 0 \\
      x + 3y +  z + 6w &=& 0 \\
      x + 2y      + 4w &=& 0
    \end{array}
  \end{equation*}
\end{ex}

\begin{ex}
  Which of the following homogeneous systems of linear equations have
  non-trivial solutions?
  \begin{enumerate}
  \item[(a)] 4 equations in 3 variables, rank 3.
  \item[(b)] 3 equations in 4 variables, rank 3.
  \item[(c)] 4 equations in 3 variables, rank 2.
  \item[(d)] 3 equations in 4 variables, rank 2.
  \end{enumerate}

  \begin{sol}
    The systems (b), (c), and (d) have non-trivial solutions, because
    the rank is less than the number of variables.
  \end{sol}
\end{ex}

\begin{ex}
  My system of equations has a solution $(x,y,z) = (1,2,4)$. The
  associated homogeneous system has basic solutions $(x,y,z) =
  (1,0,1)$ and $(x,y,z)=(0,1,-1)$. What is the general solution of my
  system of equations?

  \begin{sol}
    The general solution is $(x,y,z) = (1,2,4)+s(1,0,1)+t(0,1,-1)$, or
    equivalently, $(x,y,z) = (1+s, 2+t, 4+s-t)$, where $s$ and $t$ are
    parameters.
  \end{sol}
\end{ex}
