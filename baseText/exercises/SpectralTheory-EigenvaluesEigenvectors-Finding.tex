\begin{ex}
  Find the characteristic polynomial, eigenvalues, and basic
  eigenvectors of the matrix
  \begin{equation*}
    \begin{mymatrix}{rrr}
      0 &   3 &  -1 \\
      -2 &   4 &  -2 \\
      2 &  -3 &   3 \\
    \end{mymatrix}.
  \end{equation*}
  One eigenvalue is $1$.
\end{ex}

\begin{ex}
  Which of the following matrices have no real eigenvalue?
  \begin{equation*}
    A = \begin{mymatrix}{rr}
      1 & 1 \\
      1 & -1 \\
    \end{mymatrix},
    \quad
    B = \begin{mymatrix}{rr}
      1 & -1 \\
      1 & 1 \\
    \end{mymatrix},
    \quad
    C = \begin{mymatrix}{rr}
      0 & 1 \\
      1 & 0 \\
    \end{mymatrix}.
  \end{equation*}
\end{ex}

\begin{ex}
  Find the eigenvalues and eigenvectors of the following triangular
  matrix:
  \begin{equation*}
    \begin{mymatrix}{rrr}
      3 & 2 & 2 \\
      0 & 1 & -2 \\
      0 & 0 & -1 \\
    \end{mymatrix}.
  \end{equation*}
\end{ex}

\begin{ex}
  Is it possible for a non-zero matrix to have only $0$ as an eigenvalue?
  \vspace{1mm}
  \begin{sol}
    Yes. $\begin{mymatrix}{cc}
      0 & 1 \\
      0 & 0%
    \end{mymatrix} $ works.
  \end{sol}
\end{ex}
