\begin{ex}
  Which of the following systems of vectors are right-handed? Which
  are left-handed?
  \begin{enumerate}
  \item $\vect{i},\vect{k},\vect{j}$.
  \item $\vect{j},\vect{k},\vect{i}$.
  \item $\vect{k},\vect{i},\vect{j}$.
  \item $\vect{k},\vect{j},\vect{i}$.
  \end{enumerate}
  \begin{sol}
    (a) Left-handed.
    (b) Right-handed.
    (c) Right-handed.
    (d) Left-handed.
  \end{sol}
\end{ex}
