\begin{ex}
  Find the {\rref} of each of the following matrices $A$, and write it
  in the form $R=UA$ where $U$ is invertible.
  \begin{equation*}
    (a)~
    \begin{mymatrix}{rrr}
      1 & 3 & 1 \\
      0 & 2 & 4 \\
      2 & 6 & -2 \\
    \end{mymatrix},
    \quad
    (b)~
    \begin{mymatrix}{rr}
      0 & 1 \\
      3 & -1 \\
      2 & 6 \\
    \end{mymatrix},
    \quad
    (c)~
    \begin{mymatrix}{rrr}
      0 & 1 & -2 \\
      1 & 3 & 2 \\
    \end{mymatrix},
    \quad
    (d)~
    \begin{mymatrix}{rrr}
      1 & 1 & 0 \\
      0 & 1 & 1 \\
      1 & 0 & 1 \\
    \end{mymatrix}.
  \end{equation*}
\end{ex}
